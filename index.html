# --- 5. SERVER ---
server <- function(input, output, session) {
  
  # --- Expression Explorer Server Logic ---
  observeEvent(input$filter_species, {
    available_accs <- metadata %>% 
      filter(species %in% input$filter_species | length(input$filter_species) == 0) %>% 
      pull(accession) %>% unique() %>% sort()
    
    defaults <- c()
    if (length(input$filter_species) > 0) {
      for (s in input$filter_species) {
        accs_for_s <- metadata %>% filter(species == s) %>% pull(accession) %>% unique()
        if (length(accs_for_s) == 1) defaults <- c(defaults, accs_for_s)
        else if (s == "Tripsacum_dactyloides") defaults <- c(defaults, "Td_Pete")
        else if (s == "Zea_mays") { if ("Zm_lh287Xlh244" %in% accs_for_s) defaults <- c(defaults, "Zm_lh287Xlh244") }
        else defaults <- c(defaults, accs_for_s)
      }
    }
    updateSelectizeInput(session, "filter_accession", choices = available_accs, selected = defaults, server = TRUE)
  }, ignoreNULL = FALSE)

  target_og <- reactive({
    query_id <- str_trim(input$gene_id)
    if (query_id == "") return(NULL)
    if (str_starts(query_id, "OG")) {
      if (query_id %in% rownames(vst_counts)) return(query_id)
    } else {
      match_row <- gene_og_map %>% filter(GeneID == query_id)
      if (nrow(match_row) > 0) return(match_row$Orthogroup[1])
    }
    return(NULL)
  })
  
  output$og_id_display <- renderUI({
    og <- target_og()
    query <- str_trim(input$gene_id)
    if (is.null(og)) {
      span("ID not found or not in atlas", style = "color: #e44; font-size: 0.85em; display: block; margin-bottom: 10px;")
    } else if (str_starts(query, "OG")) {
      span(paste0(og, " \u2713"), style = "color: #282; font-weight: bold; font-size: 0.85em; display: block; margin-bottom: 10px;")
    } else {
      span(paste("Mapped Orthogroup:", og), style = "color: #282; font-weight: bold; font-size: 0.85em; display: block; margin-bottom: 10px;")
    }
  })

  # gene_data: triggered by button click OR runs once at startup (ignoreNULL = FALSE)
  gene_data <- eventReactive(input$gen_plot, {
    query_id <- str_trim(input$gene_id)
    # Fast lookup: if OG ID, use directly; otherwise use gene_og_map
    tog <- if (str_starts(query_id, "OG") && query_id %in% rownames(vst_counts)) {
      query_id
    } else if (query_id == "Zm00001eb319560" || query_id == "") {
      .default_og  # use pre-computed value, avoids slow lookup at startup
    } else {
      m <- gene_og_map$Orthogroup[gene_og_map$GeneID == query_id]
      if (length(m) > 0) m[1] else NULL
    }
    req(tog)
    
    expr_vals <- vst_counts[tog, ]
    df <- data.frame(sample_id = names(expr_vals), expression = as.numeric(expr_vals)) %>%
      inner_join(metadata, by = "sample_id")
    
    if (length(input$filter_tissue) > 0) df <- df %>% filter(tissue %in% input$filter_tissue)
    if (length(input$filter_species) > 0) df <- df %>% filter(species %in% input$filter_species)
    if (length(input$filter_accession) > 0) df <- df %>% filter(accession %in% input$filter_accession)
    df
  }, ignoreNULL = FALSE)

  y_range <- reactive({
    df <- gene_data(); req(nrow(df) > 0)
    range(df$expression, na.rm = TRUE)
  })

  active_palette <- reactive({
    gv <- input$group_var
    if (gv == "life_history") life_history_palette
    else if (gv == "tissue") tissue_type_palette
    else if (gv == "species") species_palette
    else if (gv == "accession") accession_palette_named
    else c(accession_palette_named, species_palette, tissue_type_palette, life_history_palette)
  })

  output$ribbon_plot <- renderPlotly({
    df <- gene_data(); req(nrow(df) > 0)
    
    summary_df <- df %>%
      distinct(sample_id, .keep_all = TRUE) %>%
      group_by(week, group = as.character(.data[[input$group_var]])) %>%
      summarise(mean_expr = mean(expression, na.rm = TRUE), se_expr = sd(expression, na.rm = TRUE) / sqrt(n()), .groups = "drop") %>%
      left_join(week_lookup, by = "week") %>%
      arrange(group, week) %>% filter(!is.na(mean_expr))
    
    yr <- y_range(); y_min <- yr[1]; y_max <- yr[2]; buffer <- (y_max - y_min) * 0.1
    y_lim_top <- y_max + buffer; tile_height <- (y_max - y_min) * 0.15; y_plot_top <- y_lim_top + tile_height
    
    p <- plot_ly()
    groups <- unique(summary_df$group)
    for (g in groups) {
      g_data <- summary_df %>% filter(group == g)
      color_hex <- if (g %in% names(active_palette())) active_palette()[g] else "#333333"
      
      p <- p %>% add_polygons(
        x = c(g_data$week, rev(g_data$week)), y = c(g_data$mean_expr + g_data$se_expr, rev(g_data$mean_expr - g_data$se_expr)),
        fillcolor = color_hex, opacity = 0.2, line = list(color = "transparent"), name = g, showlegend = FALSE, connectgaps = TRUE
      ) %>% add_trace(
        x = g_data$week, y = g_data$mean_expr, type = "scatter", mode = "lines+markers",
        line = list(color = color_hex, width = 3), marker = list(color = color_hex, size = 8), name = g,
        hovertext = paste0("Group: ", g, "<br>Week: ", g_data$week, "<br>Mean VST: ", round(g_data$mean_expr, 2)),
        hoverinfo = "text", connectgaps = TRUE
      )
    }
    
    shapes <- list(); annotations <- list(); month_colors <- c("July"="#228B22", "Aug"="#7CFC00", "Sept"="#ADFF2F", "Oct"="#FFFF00")
    for(i in 1:nrow(month_unique)) {
      m <- month_unique[i,]
      shapes[[i]] <- list(type = "rect", fillcolor = month_colors[m$month], line = list(color = "black", width = 1),
                          opacity = 0.4, x0 = m$start - 0.5, x1 = m$end + 0.5, y0 = y_lim_top, y1 = y_plot_top, xref = "x", yref = "y")
      annotations[[i]] <- list(x = m$mid, y = y_lim_top + (tile_height / 2), text = m$month, showarrow = FALSE, font = list(size = 12, color = "black", weight = "bold"))
    }
    
    p %>% layout(
      xaxis = list(title = "Week | Julian Day", tickvals = 1:14, ticktext = paste0("w", week_lookup$week, "|", week_lookup$julian), tickangle = 45),
      yaxis = list(title = "expression (normalized VST-counts)", range = c(y_min - buffer, y_plot_top)),
      shapes = shapes, annotations = annotations, showlegend = FALSE,
      margin = list(t = 50)
    )
  })

      coord_cartesian(ylim = c(y_min - buffer, y_max + (buffer * 2.5))) +

      return(query_id)
    } else {

      if (nrow(match_row) > 0) return(match_row$Orthogroup[1])
    }
    return(query_id)

  output$network_plot <- renderPlotly({
    req(nrow(umap_df) > 0)
    plot_data <- umap_df
    edge_data <- edges_df
    tog   <- net_target_og()           # from search box
    click <- event_data("plotly_click", source = "network_view")
    clicked_gene <- if (!is.null(click)) click$key else NULL
    
    # Active highlight: search query takes priority, then click
    active_gene <- if (!is.null(tog) && str_trim(input$net_query) != "") tog else clicked_gene
    
    is_query_active <- !is.null(active_gene)
    is_query_found  <- is_query_active && (active_gene %in% plot_data$gene)
    
    # --- Compute node display properties ---
    if (is_query_active && !is_query_found) {
      plot_data <- plot_data %>%
        mutate(display_color = "lightgrey", display_opacity = 0.15, display_size = 3)
    } else if (is_query_found) {
      plot_data <- plot_data %>%
        mutate(
          display_color = ifelse(gene == active_gene, color, "lightgrey"),
          display_opacity = ifelse(gene == active_gene, 1, 0.15),
          display_size = ifelse(gene == active_gene, 20, pmax(kME * 10, 3))
        ) %>%
        arrange(gene == active_gene)
    } else if (input$net_module != "All Modules") {
      plot_data <- plot_data %>%
        mutate(
          display_color = ifelse(module_label == input$net_module, color, "lightgrey"),
          display_opacity = ifelse(module_label == input$net_module, 0.9, 0.15),
          display_size = pmax(kME * 15, 3)
        )
    } else {
      plot_data <- plot_data %>%
        mutate(display_color = color, display_opacity = 0.8, display_size = pmax(kME * 15, 3))
    }
    
    # --- Compute edge display properties ---
    if (is_query_active && !is_query_found) {
      edge_data <- edge_data %>%
        mutate(edge_color = "lightgrey", edge_opacity = 0.03, edge_width = 0.5)
    } else if (is_query_found) {
      is_connected <- edge_data$from == active_gene | edge_data$to == active_gene
      edge_data <- edge_data %>%
        mutate(
          edge_color   = ifelse(is_connected, "red", "lightgrey"),
          edge_opacity = ifelse(is_connected, 0.8, 0.02),
          edge_width   = ifelse(is_connected, 2, 0.5)
        ) %>%
        arrange(is_connected)
    } else {
      edge_data <- edge_data %>%
        mutate(edge_color = "lightgrey", edge_opacity = 0.08, edge_width = 0.5)
    }
    
    p <- plot_ly(source = "network_view")
    
    # --- Draw edges first (so nodes appear on top) ---
    # Split into highlighted vs background to allow separate opacity control
    bg_edges <- edge_data %>% filter(edge_opacity < 0.5)
    hi_edges <- edge_data %>% filter(edge_opacity >= 0.5)
    
    if (nrow(bg_edges) > 0) {
      p <- p %>% add_segments(
        data = bg_edges,
        x = ~x0, xend = ~x1, y = ~y0, yend = ~y1,
        line = list(color = "lightgrey", width = 0.5),
        opacity = 0.08, hoverinfo = "none", showlegend = FALSE
      )
    }
    if (nrow(hi_edges) > 0) {
      p <- p %>% add_segments(
        data = hi_edges,
        x = ~x0, xend = ~x1, y = ~y0, yend = ~y1,
        line = list(color = "red", width = 2),
        opacity = 0.85, hoverinfo = "none", showlegend = FALSE
      )
    }
    
    # --- Draw nodes on top of edges ---
    p <- p %>% add_trace(
      data = plot_data,
      x = ~UMAP1, y = ~UMAP2,
      type = "scatter", mode = "markers",
      marker = list(
        color = ~display_color,
        size = ~display_size,
        opacity = ~display_opacity,
        line = list(color = "white", width = 0.5)
      ),
      text = ~paste0("OG: ", gene, "<br>Zea mays: ", Zea_mays_genes, "<br>Module: ", module_label, "<br>kME: ", round(kME, 3)),
      hoverinfo = "text",
      key = ~gene,
      showlegend = FALSE
    )
    
    p %>% layout(
      xaxis = list(title = "", showgrid = FALSE, zeroline = FALSE, showticklabels = FALSE),
      yaxis = list(title = "", showgrid = FALSE, zeroline = FALSE, showticklabels = FALSE),
      hovermode = "closest"
    )
  })

          "None"
        )
      ) %>%
      select(
        Orthogroup,
        `Zea mays Gene`,
        Module,
        `Sink/Source` = sink_source,
        `kME (Hub Score)`,
        Description
      )

  }, sanitize.text.function = function(x) x)

    clicked_gene <- click_data$key
    
    # Find all edges involving this gene
    connected <- edges_df %>%
      filter(from == clicked_gene | to == clicked_gene) %>%
      mutate(partner = ifelse(from == clicked_gene, to, from)) %>%
      select(partner, weight) %>%
      arrange(desc(weight))
    
    if (nrow(connected) == 0) return(NULL)
    
    # Build a simple lookup using column positions to avoid complex name issues
    desc_lookup <- data.frame(
      OG          = desc_df[[2]],
      zm_gene     = desc_df[[1]],
      description = as.character(desc_df[[7]]),
      stringsAsFactors = FALSE
    ) %>%
      group_by(OG) %>%
      summarise(
        `Zea mays Gene` = paste(unique(zm_gene), collapse = ", "),
        Description     = {
          descs <- unique(description[description != "" & !is.na(description)])
          if (length(descs) == 0) "No description available" else paste(descs, collapse = " / ")
        },
        .groups = "drop"
      )
    
    connected %>%
      left_join(desc_lookup, by = c("partner" = "OG")) %>%
      mutate(
        Description     = ifelse(is.na(Description), "No description available", Description),
        `Zea mays Gene` = ifelse(is.na(`Zea mays Gene`), "None", `Zea mays Gene`),
        weight          = round(weight, 4)
      ) %>%
      rename(
        `Partner OG` = partner,
        `TOM Weight` = weight
      )
  })
